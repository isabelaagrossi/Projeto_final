AACPD

data_sev <- read_xlsx("dados_in_vivo.xlsx", 3)

data_sev$tempo <- as.numeric(as.character(data_sev$tempo))

a1 <- data_sev |> 
  group_by(trat, rep) |> 
  summarise(aacpd = AUDPC(tempo, severidade))
a1


m_a1 <- lm(aacpd ~ trat + factor(rep),
